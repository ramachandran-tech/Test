-- ============================================================
-- WORKFORCE ROW-LEVEL SECURITY VIEW
-- ============================================================
-- Version:  FINAL (Single scan, Delta Cache + Photon optimized)
-- Platform: Databricks / Delta Lake, Runtime 13.3+
--
-- Key design:
--   Single workforce reference → one scan → delta cache works
--   ARRAY_CONTAINS for membership (JOIN ON)
--   ARRAY_POSITION + SLICE for 4-level check (inline, no depth CTE)
--   Wildcard separated → no match-all poisoning optimizer
--   Subquery EXPLODE → Photon native
--   BROADCAST → no shuffle on workforce
--
-- Delta cache behavior:
--   Run 1: sequential workforce scan → files cached
--   Run 2: 100% from delta cache → near-zero disk I/O
--
-- Infrastructure: ZERO additional tables
-- ============================================================


CREATE OR REPLACE VIEW workforce_secured AS

WITH



-- ============================================================
-- STEP 1: Current user's roles
-- Dashboard may push role filter → 1 row
-- Without filter → all applicable roles for this user
-- ============================================================
user_roles AS (
    SELECT DISTINCT
        ur.email,
        ur.role,
        ur.description AS role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        CASE WHEN mr.role_name IS NOT NULL THEN TRUE ELSE FALSE END AS is_manager
    FROM userrole_tab ur
    LEFT JOIN manager_roles mr ON ur.role = mr.role_name
    WHERE ur.email = current_user()
      AND (
          ur.role IN (SELECT role_name FROM role_exact)
          OR EXISTS (
              SELECT 1 FROM role_prefixes rp
              WHERE ur.role LIKE CONCAT(rp.prefix, '%')
          )
      )
),


-- ============================================================
-- STEP 2: Explode managementview (Photon-native subquery syntax)
-- Result: ~5-25 rows (tiny)
-- ============================================================
user_nodes_exploded AS (
    SELECT
        ur.role,
        ur.role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        ur.is_manager,
        mv_node
    FROM user_roles ur,
    LATERAL (SELECT EXPLODE(ur.managementview) AS mv_node) mv
),


-- ============================================================
-- STEP 3a: Wildcard nodes (mv_node = '*')
-- Separated so '*' doesn't poison JOIN optimizer.
-- CTE is EMPTY when no wildcard role → zero cost.
-- ============================================================
user_nodes_wildcard AS (
    SELECT
        role, role_description, managementview,
        country, sbmod, division, jobgroup, companycode,
        is_manager
    FROM user_nodes_exploded
    WHERE mv_node = '*'
),


-- ============================================================
-- STEP 3b: Specific nodes — NO depth discovery scan
-- Just filters out wildcard. Depth computed inline in STEP 4b.
-- ============================================================
user_nodes_specific AS (
    SELECT
        role, role_description, managementview,
        country, sbmod, division, jobgroup, companycode,
        is_manager, mv_node
    FROM user_nodes_exploded
    WHERE mv_node != '*'
),


-- ============================================================
-- STEP 4a: Wildcard path — full workforce access
-- SAFE: CTE is EMPTY when no wildcard role → zero workforce scan.
-- ============================================================
wildcard_rows AS (
    SELECT
        w.*,
        uw.role,
        uw.role_description
    FROM /*+ BROADCAST(uw) */ user_nodes_wildcard uw
    INNER JOIN workforce w
        ON TRUE
    WHERE
            (uw.country[0]     = '*' OR ARRAY_CONTAINS(uw.country,     w.country))
        AND (uw.sbmod[0]       = '*' OR ARRAY_CONTAINS(uw.sbmod,       w.sbmod))
        AND (uw.division[0]    = '*' OR ARRAY_CONTAINS(uw.division,    w.division))
        AND (uw.jobgroup[0]    = '*' OR ARRAY_CONTAINS(uw.jobgroup,    w.jobgroup))
        AND (uw.companycode[0] = '*' OR ARRAY_CONTAINS(uw.companycode, w.companycode))
),


-- ============================================================
-- STEP 4b: Specific path — SINGLE workforce scan
--
-- This is the ONLY workforce reference for non-wildcard roles.
-- One scan → delta cache caches → rerun reads from cache.
--
-- JOIN ON:
--   ARRAY_CONTAINS(level_array, mv_node) → membership check
--
-- WHERE (management view):
--   is_manager = FALSE → skip everything (1 boolean)
--   is_manager = TRUE  →
--     ARRAY_POSITION(level_array, mv_node) finds depth INLINE
--     No separate depth discovery scan needed
--     SLICE(level_array, depth, 5) extracts 4-level window
--     ARRAY_CONTAINS(window, managementview) checks presence
--
-- Safety chain:
--   Membership JOIN guarantees mv_node IS in level_array
--   → ARRAY_POSITION always returns > 0 for joined rows
--   → SLICE always has valid start index
--   Non-manager: is_manager=FALSE short-circuits before any of this
--   Extra guard: ARRAY_POSITION > 0 for defensive safety
-- ============================================================
specific_rows AS (
    SELECT DISTINCT
        w.*,
        us.role,
        us.role_description
    FROM /*+ BROADCAST(us) */ user_nodes_specific us
    INNER JOIN workforce w
        ON ARRAY_CONTAINS(
            ARRAY(
                w.level1_node,  w.level2_node,  w.level3_node,
                w.level4_node,  w.level5_node,  w.level6_node,
                w.level7_node,  w.level8_node,  w.level9_node,
                w.level10_node, w.level11_node, w.level12_node,
                w.level13_node, w.level14_node
            ),
            us.mv_node
        )
    WHERE
        -- Management view 4-level restriction
        -- Non-manager: 1 boolean check, skip everything
        -- Manager: depth discovered inline via ARRAY_POSITION
        (
            us.is_manager = FALSE
            OR (
                ARRAY_POSITION(
                    ARRAY(
                        w.level1_node,  w.level2_node,  w.level3_node,
                        w.level4_node,  w.level5_node,  w.level6_node,
                        w.level7_node,  w.level8_node,  w.level9_node,
                        w.level10_node, w.level11_node, w.level12_node,
                        w.level13_node, w.level14_node
                    ),
                    us.mv_node
                ) > 0
                AND ARRAY_CONTAINS(
                    SLICE(
                        ARRAY(
                            w.level1_node,  w.level2_node,  w.level3_node,
                            w.level4_node,  w.level5_node,  w.level6_node,
                            w.level7_node,  w.level8_node,  w.level9_node,
                            w.level10_node, w.level11_node, w.level12_node,
                            w.level13_node, w.level14_node
                        ),
                        ARRAY_POSITION(
                            ARRAY(
                                w.level1_node,  w.level2_node,  w.level3_node,
                                w.level4_node,  w.level5_node,  w.level6_node,
                                w.level7_node,  w.level8_node,  w.level9_node,
                                w.level10_node, w.level11_node, w.level12_node,
                                w.level13_node, w.level14_node
                            ),
                            us.mv_node
                        ),
                        5
                    ),
                    w.managementview
                )
            )
        )

        -- Dimensional
        AND (us.country[0]     = '*' OR ARRAY_CONTAINS(us.country,     w.country))
        AND (us.sbmod[0]       = '*' OR ARRAY_CONTAINS(us.sbmod,       w.sbmod))
        AND (us.division[0]    = '*' OR ARRAY_CONTAINS(us.division,    w.division))
        AND (us.jobgroup[0]    = '*' OR ARRAY_CONTAINS(us.jobgroup,    w.jobgroup))
        AND (us.companycode[0] = '*' OR ARRAY_CONTAINS(us.companycode, w.companycode))
),


-- ============================================================
-- STEP 5: Combine paths
-- Only ONE path active at runtime.
-- Empty path = zero cost, optimizer skips.
-- ============================================================
all_secured_rows AS (
    SELECT * FROM wildcard_rows
    UNION ALL
    SELECT * FROM specific_rows
),


-- ============================================================
-- STEP 6: Column definitions — ALL in ONE place
-- Both pure derivations and role-dependent masking.
-- Evaluated only on rows that passed security (~20K).
-- Change any column → change ONLY here.
-- ============================================================
final_output AS (
    SELECT
        *

    FROM all_secured_rows r
)

SELECT * FROM final_output;
