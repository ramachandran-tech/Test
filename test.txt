WITH
role_prefixes AS (
    SELECT prefix FROM (VALUES
        ('XXAHR6A'), ('XXAHR6B'), ('XXGBP0A'), ('XXLBP0A'),
        ('XXINA0A'), ('XXLCB4A'), ('XXMANDA')
    ) AS t(prefix)
),

role_exact AS (
    SELECT role_name FROM (VALUES
        ('XXCBP0AC'), ('XXGCB0AJ'), ('XX8001RX'), ('XXCCB4AA'),
        ('XXCRD2AX'), ('XXCTM3AX'), ('XXGC04AA'), ('XXSMA0AX'),
        ('XXSW1AX'), ('DEVELOPER'), ('ANALYST')
    ) AS t(role_name)
),

manager_roles AS (
    SELECT role_name FROM (VALUES
        ('XXAHR6A1'),
        ('XXAHR6A2'),
        ('XXMANDAX')
    ) AS t(role_name)
),

user_roles AS (
    SELECT DISTINCT
        ur.email,
        ur.role,
        ur.description AS role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        CASE WHEN mr.role_name IS NOT NULL THEN TRUE ELSE FALSE END AS is_manager
    FROM securedbt_hr.hr_sec_userrole ur
    LEFT JOIN manager_roles mr ON ur.role = mr.role_name
    WHERE ur.email = current_user()
      AND (
          ur.role IN (SELECT role_name FROM role_exact)
          OR EXISTS (
              SELECT 1 FROM role_prefixes rp
              WHERE startswith(ur.role, rp.prefix)
          )
      )
),

user_nodes AS (
    SELECT
        ur.role,
        ur.role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        ur.is_manager,
        mv_node
    FROM user_roles ur,
    LATERAL (SELECT EXPLODE(ur.managementview) AS mv_node) mv
),

all_secured_rows AS (
    SELECT DISTINCT
        w.*,
        un.role,
        un.role_description
    FROM /*+ BROADCAST(un) */ user_nodes un
    INNER JOIN workforce w
        ON (
            un.mv_node = '*'
            OR ARRAY_CONTAINS(
                ARRAY(
                    w.level1_node,  w.level2_node,  w.level3_node,
                    w.level4_node,  w.level5_node,  w.level6_node,
                    w.level7_node,  w.level8_node,  w.level9_node,
                    w.level10_node, w.level11_node, w.level12_node,
                    w.level13_node, w.level14_node
                ),
                un.mv_node
            )
        )
    WHERE
        (
            un.is_manager = FALSE
            OR un.managementview[0] = '*'
            OR (
                ARRAY_POSITION(
                    ARRAY(
                        w.level1_node,  w.level2_node,  w.level3_node,
                        w.level4_node,  w.level5_node,  w.level6_node,
                        w.level7_node,  w.level8_node,  w.level9_node,
                        w.level10_node, w.level11_node, w.level12_node,
                        w.level13_node, w.level14_node
                    ),
                    un.mv_node
                ) > 0
                AND ARRAY_CONTAINS(
                    SLICE(
                        ARRAY(
                            w.level1_node,  w.level2_node,  w.level3_node,
                            w.level4_node,  w.level5_node,  w.level6_node,
                            w.level7_node,  w.level8_node,  w.level9_node,
                            w.level10_node, w.level11_node, w.level12_node,
                            w.level13_node, w.level14_node
                        ),
                        ARRAY_POSITION(
                            ARRAY(
                                w.level1_node,  w.level2_node,  w.level3_node,
                                w.level4_node,  w.level5_node,  w.level6_node,
                                w.level7_node,  w.level8_node,  w.level9_node,
                                w.level10_node, w.level11_node, w.level12_node,
                                w.level13_node, w.level14_node
                            ),
                            un.mv_node
                        ),
                        5
                    ),
                    w.managementview
                )
            )
        )
        AND (un.country[0]     = '*' OR ARRAY_CONTAINS(un.country,     w.country))
        AND (un.sbmod[0]       = '*' OR ARRAY_CONTAINS(un.sbmod,       w.sbmod))
        AND (un.division[0]    = '*' OR ARRAY_CONTAINS(un.division,    w.division))
        AND (un.jobgroup[0]    = '*' OR ARRAY_CONTAINS(un.jobgroup,    w.jobgroup))
        AND (un.companycode[0] = '*' OR ARRAY_CONTAINS(un.companycode, w.companycode))
),

final_output AS (
    SELECT
        *
    FROM all_secured_rows r
)

SELECT * FROM final_output;
