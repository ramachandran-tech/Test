-- ============================================================
-- WORKFORCE ROW-LEVEL SECURITY VIEW
-- ============================================================
-- Version:  FINAL (single path, no branches)
-- Platform: Databricks / Delta Lake
--
-- Security model: deny-all-by-default
--
-- Design:
--   ONE query path handles everything:
--     Prima, wildcard, non-manager, manager — all inline
--     is_manager = FALSE short-circuits manager logic
--     array[0] = '*' short-circuits wildcard checks
--     No UNION ALL, no separate branches
--     Single workforce scan
--
-- Infrastructure: ZERO additional tables
-- ============================================================


CREATE OR REPLACE VIEW workforce_secured AS

WITH

-- ============================================================
-- ROLE FILTER CONFIGURATION
-- Only section that changes per view.
-- ============================================================
role_prefixes AS (
    SELECT prefix FROM (VALUES 
        ('XXAHR6A'), ('XXAHR6B'), ('XXGBP0A'), ('XXLBP0A'),
        ('XXINA0A'), ('XXLCB4A'), ('XXMANDA')
    ) AS t(prefix)
),

role_exact AS (
    SELECT role_name FROM (VALUES 
        ('XXCBP0AC'), ('XXGCB0AJ'), ('XX8001RX'), ('XXCCB4AA'),
        ('XXCRD2AX'), ('XXCTM3AX'), ('XXGC04AA'), ('XXSMA0AX'),
        ('XXSW1AX'), ('DEVELOPER'), ('ANALYST')
    ) AS t(role_name)
),

manager_roles AS (
    SELECT role_name FROM (VALUES 
        ('XXAHR6A1'),
        ('XXAHR6A2'),
        ('XXMANDAX')
    ) AS t(role_name)
),


-- ============================================================
-- STEP 1: Current user's roles
-- Dashboard may push role filter → 1 row
-- Without filter → all applicable roles
-- ============================================================
user_roles AS (
    SELECT DISTINCT
        ur.email,
        ur.role,
        ur.description AS role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        CASE WHEN mr.role_name IS NOT NULL THEN TRUE ELSE FALSE END AS is_manager
    FROM securedbt_hr.hr_sec_userrole ur
    LEFT JOIN manager_roles mr ON ur.role = mr.role_name
    WHERE ur.email = current_user()
      AND (
          ur.role IN (SELECT role_name FROM role_exact)
          OR EXISTS (
              SELECT 1 FROM role_prefixes rp
              WHERE ur.role LIKE CONCAT(rp.prefix, '%')
          )
      )
),


-- ============================================================
-- STEP 2: Explode managementview + discover depth
--
-- Explodes ~5 nodes per role. Depth discovered from workforce.
-- '*' nodes: LEFT JOIN produces NULL depth (correct, never used)
-- Non-manager nodes: depth discovered but never used (harmless)
-- Manager nodes: depth used in 4-level range guard
--
-- Result: ~5-25 rows depending on role count
-- ============================================================
user_nodes AS (
    SELECT 
        ur.role,
        ur.role_description,
        ur.managementview,
        ur.country,
        ur.sbmod,
        ur.division,
        ur.jobgroup,
        ur.companycode,
        ur.is_manager,
        mv_node,
        MIN(CASE
            WHEN w.level1_node  = mv_node THEN 1
            WHEN w.level2_node  = mv_node THEN 2
            WHEN w.level3_node  = mv_node THEN 3
            WHEN w.level4_node  = mv_node THEN 4
            WHEN w.level5_node  = mv_node THEN 5
            WHEN w.level6_node  = mv_node THEN 6
            WHEN w.level7_node  = mv_node THEN 7
            WHEN w.level8_node  = mv_node THEN 8
            WHEN w.level9_node  = mv_node THEN 9
            WHEN w.level10_node = mv_node THEN 10
            WHEN w.level11_node = mv_node THEN 11
            WHEN w.level12_node = mv_node THEN 12
            WHEN w.level13_node = mv_node THEN 13
            WHEN w.level14_node = mv_node THEN 14
        END) AS mv_node_depth
    FROM user_roles ur
    LATERAL VIEW EXPLODE(ur.managementview) mv AS mv_node
    LEFT JOIN workforce w
        ON mv_node != '*'
        AND (
               w.level1_node  = mv_node OR w.level2_node  = mv_node
            OR w.level3_node  = mv_node OR w.level4_node  = mv_node
            OR w.level5_node  = mv_node OR w.level6_node  = mv_node
            OR w.level7_node  = mv_node OR w.level8_node  = mv_node
            OR w.level9_node  = mv_node OR w.level10_node = mv_node
            OR w.level11_node = mv_node OR w.level12_node = mv_node
            OR w.level13_node = mv_node OR w.level14_node = mv_node
        )
    GROUP BY
        ur.role, ur.role_description,
        ur.managementview, ur.country, ur.sbmod,
        ur.division, ur.jobgroup, ur.companycode,
        ur.is_manager, mv_node
),


-- ============================================================
-- STEP 3: Single scan — all security logic in one pass
--
-- Short-circuit evaluation order per workforce row:
--
-- MEMBERSHIP (JOIN ON):
--   mv_node = '*' → TRUE instantly, skip 14 level checks
--   mv_node = w.level1_node → TRUE, stop
--   mv_node = w.level2_node → TRUE, stop
--   ... up to level14
--
-- MANAGEMENT VIEW (WHERE):
--   is_manager = FALSE → TRUE → skip entire block (1 comparison)
--   is_manager = TRUE:
--     managementview[0] = '*' → TRUE → skip range checks (1 comparison)
--     managementview[0] != '*':
--       14 range-guarded branches, ~5 evaluate, rest short-circuit
--
-- DIMENSIONAL (WHERE):
--   array[0] = '*' → TRUE → skip value check (1 comparison per column)
--   array[0] != '*' → ARRAY_CONTAINS checks value
--
-- Cost per row:
--   Prima/wildcard:     ~7 comparisons (all short-circuit)
--   Non-manager:        ~20-70 comparisons (membership + dimensional)
--   Manager specific:   ~40-90 comparisons (membership + mgmt view + dimensional)
-- ============================================================
all_secured_rows AS (
    SELECT DISTINCT
        w.*,
        un.role,
        un.role_description
    FROM user_nodes un
    INNER JOIN workforce w
        ON (
            -- Membership
            un.mv_node = '*'
            OR un.mv_node = w.level1_node
            OR un.mv_node = w.level2_node
            OR un.mv_node = w.level3_node
            OR un.mv_node = w.level4_node
            OR un.mv_node = w.level5_node
            OR un.mv_node = w.level6_node
            OR un.mv_node = w.level7_node
            OR un.mv_node = w.level8_node
            OR un.mv_node = w.level9_node
            OR un.mv_node = w.level10_node
            OR un.mv_node = w.level11_node
            OR un.mv_node = w.level12_node
            OR un.mv_node = w.level13_node
            OR un.mv_node = w.level14_node
        )
    WHERE
        -- Management view 4-level restriction
        -- Non-manager: is_manager=FALSE → TRUE → skip (1 comparison)
        -- Manager wildcard: [0]='*' → TRUE → skip (2 comparisons)
        -- Manager specific: evaluate range guards
        (
            un.is_manager = FALSE
            OR un.managementview[0] = '*'
            OR (
                   (w.managementview = w.level1_node  AND 1  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level2_node  AND 2  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level3_node  AND 3  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level4_node  AND 4  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level5_node  AND 5  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level6_node  AND 6  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level7_node  AND 7  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level8_node  AND 8  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level9_node  AND 9  BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level10_node AND 10 BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level11_node AND 11 BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level12_node AND 12 BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level13_node AND 13 BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
                OR (w.managementview = w.level14_node AND 14 BETWEEN un.mv_node_depth AND un.mv_node_depth + 4)
            )
        )

        -- Dimensional
        AND (un.country[0]     = '*' OR ARRAY_CONTAINS(un.country,     w.country))
        AND (un.sbmod[0]       = '*' OR ARRAY_CONTAINS(un.sbmod,       w.sbmod))
        AND (un.division[0]    = '*' OR ARRAY_CONTAINS(un.division,    w.division))
        AND (un.jobgroup[0]    = '*' OR ARRAY_CONTAINS(un.jobgroup,    w.jobgroup))
        AND (un.companycode[0] = '*' OR ARRAY_CONTAINS(un.companycode, w.companycode))
),


-- ============================================================
-- STEP 4: Column definitions — ALL in ONE place
-- Derivations + role-dependent masking
-- Evaluated only on rows that passed security
-- Change any column → change ONLY here
-- ============================================================
final_output AS (
    SELECT
        r.ipin,
        r.snapshotdate,
        r.role,
        r.role_description,

        -- Pass-through columns
        r.employee_name,
        r.hire_date,
        r.department,
        r.country,
        r.division,

        -- Pure derivations (role-independent)
        CASE 
            WHEN r.fte < 0.5 THEN 'Part Time'
            WHEN r.fte < 1.0 THEN 'Reduced Hours'
            ELSE 'Full Time'
        END AS fte_category,

        COALESCE(r.cost_center, r.department_code, 'UNKNOWN') AS resolved_cost_center,

        CASE 
            WHEN r.termination_date IS NOT NULL 
             AND r.termination_date <= r.snapshotdate 
            THEN 'Terminated'
            WHEN r.leave_start IS NOT NULL AND r.leave_end IS NULL 
            THEN 'On Leave'
            ELSE 'Active'
        END AS derived_status,

        -- Role-dependent masking
        CASE 
            WHEN r.role IN ('XXAHR6A1', 'XXLBP0A1') THEN ''
            ELSE r.gender 
        END AS gender,

        CASE 
            WHEN r.role IN ('XXAHR6A1', 'XXLBP0A1', 'ANALYST') THEN NULL
            ELSE r.salary 
        END AS salary,

        CASE 
            WHEN r.role NOT IN ('XXGCB0AJ', 'DEVELOPER') THEN NULL
            ELSE r.date_of_birth 
        END AS date_of_birth,

        CASE 
            WHEN r.role LIKE 'XXAHR6%' THEN r.personal_email
            ELSE ''
        END AS personal_email

        -- ... remaining columns ...

    FROM all_secured_rows r
)

SELECT * FROM final_output;
